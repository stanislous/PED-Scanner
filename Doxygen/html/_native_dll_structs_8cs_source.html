816  public PeHeaderReader(string filePath)
817  {
818  if (filePath != null)
819  {
820  // Read in the DLL or EXE and get the timestamp
821  using (FileStream stream = new FileStream(filePath, System.IO.FileMode.Open, System.IO.FileAccess.Read))
822  {
823  BinaryReader reader = new BinaryReader(stream);
824  dosHeader = FromBinaryReader<IMAGE_DOS_HEADER>(reader);
825 
826  // Add 4 bytes to the offset
827  stream.Seek(dosHeader.e_lfanew, SeekOrigin.Begin);
828 
829  UInt32 ntHeadersSignature = reader.ReadUInt32();
830  fileHeader = FromBinaryReader<IMAGE_FILE_HEADER>(reader);
831  if (this.Is32BitHeader)
832  {
833  optionalHeader32 = FromBinaryReader<IMAGE_OPTIONAL_HEADER32>(reader);
834  }
835  else
836  {
837  optionalHeader64 = FromBinaryReader<IMAGE_OPTIONAL_HEADER64>(reader);
838  }
839  // imageNTHeader64 = FromBinaryReader<PeHeaderReader.IMAGE_NT_HEADERS64>(reader);
840  imageSectionHeaders = new IMAGE_SECTION_HEADER[fileHeader.NumberOfSections];
841  for (int headerNo = 0; headerNo < imageSectionHeaders.Length; ++headerNo)
842  {
843  imageSectionHeaders[headerNo] = FromBinaryReader<IMAGE_SECTION_HEADER>(reader);
844  }
845 
846  }
847  }
848  }
849  public static T FromBinaryReader<T>(BinaryReader reader)
850  {
851  // Read in a byte array
852  byte[] bytes = reader.ReadBytes(Marshal.SizeOf(typeof(T)));
853 
854  // Pin the managed memory while, copy it out the data, then unpin it
855  GCHandle handle = GCHandle.Alloc(bytes, GCHandleType.Pinned);
856  T theStructure = (T)Marshal.PtrToStructure(handle.AddrOfPinnedObject(), typeof(T));
857  handle.Free();
858 
859  return theStructure;
860  }
