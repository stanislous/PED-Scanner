836  public PeHeaderReader(string filePath)
837  {
838  if (filePath != null)
839  {
840  // Read in the DLL or EXE and get the timestamp
841  using (FileStream stream = new FileStream(filePath, System.IO.FileMode.Open, System.IO.FileAccess.Read))
842  {
843  BinaryReader reader = new BinaryReader(stream);
844  dosHeader = FromBinaryReader<IMAGE_DOS_HEADER>(reader);
845 
846  // Add 4 bytes to the offset
847  stream.Seek(dosHeader.e_lfanew, SeekOrigin.Begin);
848 
849  UInt32 ntHeadersSignature = reader.ReadUInt32();
850  fileHeader = FromBinaryReader<IMAGE_FILE_HEADER>(reader);
851  if (this.Is32BitHeader)
852  {
853  optionalHeader32 = FromBinaryReader<IMAGE_OPTIONAL_HEADER32>(reader);
854  }
855  else
856  {
857  optionalHeader64 = FromBinaryReader<IMAGE_OPTIONAL_HEADER64>(reader);
858  }
859  // imageNTHeader64 = FromBinaryReader<PeHeaderReader.IMAGE_NT_HEADERS64>(reader);
860  imageSectionHeaders = new IMAGE_SECTION_HEADER[fileHeader.NumberOfSections];
861  for (int headerNo = 0; headerNo < imageSectionHeaders.Length; ++headerNo)
862  {
863  imageSectionHeaders[headerNo] = FromBinaryReader<IMAGE_SECTION_HEADER>(reader);
864  }
865 
866  }
867  }
868  }
869  public static T FromBinaryReader<T>(BinaryReader reader)
870  {
871  // Read in a byte array
872  byte[] bytes = reader.ReadBytes(Marshal.SizeOf(typeof(T)));
873 
874  // Pin the managed memory while, copy it out the data, then unpin it
875  GCHandle handle = GCHandle.Alloc(bytes, GCHandleType.Pinned);
876  T theStructure = (T)Marshal.PtrToStructure(handle.AddrOfPinnedObject(), typeof(T));
877  handle.Free();
878 
879  return theStructure;
880  }
